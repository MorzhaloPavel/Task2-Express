const User = require('./user.model')
const DBTASK = require('../tasks/tasks.memory.repository')

let DBUsers = []
/**
 * Get all users
 * @returns {Array} Array of objects(user)
 */
const getAll = () => DBUsers;

/**
 * Get by id user
 * @param {string} id The User id
 * @returns {Object} The user
 */
const get = async id => {
  const user = await DBUsers.filter(user => user.id === id)
  return user[0]
}

/**
 * Add new user in DBUsers
 * @param {User} user The User object
 * @returns {Object} The new user
 */
const save = async user => {
  newUser = new User(user)
  DBUsers = [...DBUsers, newUser]
  return newUser
}

const remove = async id => {
  DBUsers = await DBUsers.filter(user => user.id !== id)
  DBTASK.removeUserTasks(id)
}

const update = async (id, user) => {
  DBUsers = DBUsers.map(user => {
    if (user.id === id) {
      return {id, ...user};
    }
    return user
  })
  return get(id)
}

module.exports = { getAll, get, save, remove, update };
